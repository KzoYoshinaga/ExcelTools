Private Function test()


    Dim t As New table

    Sheets("pile").Cells.Clear
    Call t. _
          import(cell("A", "A1")). _
          skipPile(t.import(cell("B", "A1")), 4). _
          export(cell("pile", "A1")). _
          terminate

    Sheets("zip").Cells.Clear
    Call t. _
          import(cell("A", "A1")). _
          skipZip(t.import(cell("B", "A1")), 2). _
          export(cell("zip", "A1")). _
          terminate

    Sheets("test").Cells.Clear
    Call t. _
          import(cell("zip", "A1")). _
          insertColumn(t.import(cell("pile", "A1")), 2). _
          export(cell("test", "A1")). _
          terminate
End Function
