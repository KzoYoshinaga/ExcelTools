' Tuples3 **********************************************

' 新しいインスタンスが生成されること
Private Function createNewInstance()
    ' SetTestName
    testName = "createNewInstance()"

    ' Arrange
    Dim values As New collection
    values.add newElement("a")

    Dim t1 As New Tuple
    t1.setValues = values
    Dim t2 As New Tuple
    t2.setValues = values
    Dim t3 As New Tuple
    t3.setValues = values

    ' Do
    Dim ts3 As New Tuples3
    ts3.setLeft = t1
    ts3.setMiddle = t2
    ts3.setRight = t3

    ' Verification
    result = equals(ts3.toString, "Tuples3(Tuple(a), Tuple(a), Tuple(a))")
    verify result, testName
End Function
