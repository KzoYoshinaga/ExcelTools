' Tuples3 **********************************************

' 新しいインスタンスが生成されること
Private Function createNewInstance()
    ' SetTestName
    testName = "createNewInstance()"

    ' Arrange
    Dim values As New collection
    values.add "a"
     
    Dim t1 As New Tuple
    t1.setValues = values
    Dim t2 As New Tuple
    t2.setValues = values
    Dim t3 As New Tuple
    t3.setValues = values
       
    ' Do
    Dim ts3 As New Tuples3
    ts3.setLeft = t1
    ts3.setMiddle = t2
    ts3.setRight = t3
       
    ' Verification
    result = equals(ts3.toString, "Tuples3(Tuple(a), Tuple(a), Tuple(a))")
    verify result, testName
End Function

' 左タプルが取り出せること
Private Function getLeft()
    ' SetTestName
    testName = "getLeft()"

    ' Arrange
    Dim f As New Factory
    
    Dim values1 As New collection
    values1.add "a"
    Dim values2 As New collection
    values2.add "b"
    Dim values3 As New collection
    values3.add "c"
    
    Dim ts3 As New Tuples3
    ts3.setLeft = f.newTuple(values1)
    ts3.setMiddle = f.newTuple(values2)
    ts3.setRight = f.newTuple(values3)
   
    Dim r As Tuple
    Set r = ts3.getLeft
    
    ' Verification
    result = equals(r.toString, "Tuple(a)")
    verify result, testName
End Function
